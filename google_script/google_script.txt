function doPost(e) {
  var data = Utilities.base64Decode(e.parameters.data);
  var nombreArchivo = Utilities.formatDate(new Date(), "GMT+7", "yyyyMMdd_HHmmss")+".jpg";
  var blob = Utilities.newBlob(data, e.parameters.mimetype, nombreArchivo );
    
   // Simpan foto ke Google Drive
  var folder, folders = DriveApp.getFoldersByName("CAMERA1");
  if (folders.hasNext()) {
    folder = folders.next();
  } else {
    folder = DriveApp.createFolder("CAMERA1");
  }
  var file = folder.createFile(blob); 
  return ContentService.createTextOutput('SUKSESS!!')
}
